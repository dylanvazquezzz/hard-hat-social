    ```typescript
    import { useRouter } from 'next/navigation'

    function PendingReviewMessage() {
      return (
        <div className="mx-auto max-w-2xl px-4 py-20 text-center">
          <h1 className="text-2xl font-bold text-slate-100">Application Under Review</h1>

            </a>{' '}
            for community posts and Q&amp;A.
          </p>
        </div>
      )

    export default function ContractorsLayout({ children }: { children: React.ReactNode }) {
      const router = useRouter()
      const [status, setStatus] = useState<'loading' | 'pending' | 'ok'>('loading')

      useEffect(() => {
        supabase.auth.getSession().then(async ({ data: { session } }) => {
          if (!session) {
            router.replace('/auth')
            return
          }
          const { data: app } = await supabase
            .from('applications')
            .select('status')
            .eq('user_id', session.user.id)
            .eq('status', 'pending')
            .maybeSingle()
          setStatus(app ? 'pending' : 'ok')
        })
      }, [router])

      if (status === 'loading') return null
      if (status === 'pending') return <PendingReviewMessage />
      return <>{children}</>
    }
    ```

    KEY RULES:
    - Query `applications` table (NOT `contractors`) — pending users have no `contractors` row
    - Use `.maybeSingle()` (NOT `.single()`) — returns null when no matching row, no error thrown
    - No-application users (status 'ok') get full access — only users with an active pending application row are blocked
    - The `/explore` link in the message goes to `/explore` (not filtered) — the explore page itself controls what's shown
